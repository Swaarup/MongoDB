LOAD CSV WITH HEADERS FROM 'https://raw.githubusercontent.com/Swaarup/MongoDB/main/customer.csv' AS row
MERGE (c:customer{_id:row._id, createdAt:row.createdAt, customerAddress:row.customeraddress, dob: row.dob, email: row.email, firstName:row.firstName, lastName:row.lastName, status: row.status});
